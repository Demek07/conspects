## 3. –°–∏–Ω—Ç–∞–∫—Å–∏—Å –≥–µ—Ç—Ç–µ—Ä–æ–≤ –∏ —Å–µ—Ç—Ç–µ—Ä–æ–≤ üîß

–î–∞–≤–∞–π—Ç–µ —Ä–∞–∑–±–µ—Ä–µ–º—Å—è —Å —Å–∏–Ω—Ç–∞–∫—Å–∏—Å–æ–º –≥–µ—Ç—Ç–µ—Ä–æ–≤ –∏ —Å–µ—Ç—Ç–µ—Ä–æ–≤. –°–æ–∑–¥–∞–¥–∏–º –ø—Ä–æ—Å—Ç–æ–π –∫–ª–∞—Å—Å `Person`.

```python
class Person:
    def __init__(self, name, age):
        self._name = name
        self._age = age

    # Getter for name
    @property
    def name(self):
        return self._name

    # Setter for name
    @name.setter
    def name(self, value):
        if len(value) > 0:
            self._name = value
        else:
            raise ValueError("Name should not be empty")

    # Getter for age
    @property
    def age(self):
        return self._age

    # Setter for age
    @age.setter
    def age(self, value):
        if value > 0:
            self._age = value
        else:
            raise ValueError("Age should be greater than 0")
```

def overload(function):
    def wrapper(*args, **kwargs):
        if function.__name__ == "calculate_area":
            if len(args) == 1:
                return function(args[0], 0)
            elif len(args) == 2:
                return function(*args)
    return wrapper


class Shape:
    @overload
    def calculate_area(self, side_length, height):
        if height == 0:
            return side_length ** 2
        else:
            return side_length * height / 2


square = Shape()
triangle = Shape()

print(square.calculate_area(4))       # Output: 16
